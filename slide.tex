  \end{itemize}

\begin{frame}
  \frametitle{血流の改善が炎症を抑制する可能性}
  \fbox{ \onslide<2>{$\alpha_1$遮断} }薬やホスホジエステラーゼ5阻害薬は\\
  平滑筋を弛緩させ、尿道の狭窄を改善する
  \vfill
  →血流を改善させ、酸化ストレスを軽減、\\
  炎症を抑制する可能性も考えられる
\end{frame}
